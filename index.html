// index.html 内のボタン処理 function goToStripe(plan) { if (plan === 'travel') { // $24.99 決済へ window.location.href = 'https://buy.stripe.com/7sY28t7Rje4UgzZ1DNasg0q'; } else if (plan === 'basic') { // $2.99 決済へ window.location.href = 'https://buy.stripe.com/bJedRbefH5yogzZfuDasg09'; } } // 決済完了後、translate.html側でプランを判定するロジック（BEE実装） window.onload = function() { const params = new URLSearchParams(window.location.search); const plan = params.get('plan'); if (plan === 'travel') { localStorage.setItem('user_rank', 'traveler'); showTravelBriefing(); // 旅行者専用の「笑顔」を守る画面へ } else if (plan === 'basic') { localStorage.setItem('user_rank', 'standard'); unlockUnlimited(); // 日常の制限解除へ } }
